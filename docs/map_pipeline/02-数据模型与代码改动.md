### 3a: import_data() 主流程调整

```python
def import_data(self, geojson, bounds=None, scale=10000.0, region_id="imported",
                segment_length=80.0, building_snap_distance=30.0) -> Region:

    features = geojson.get("features", [])
    if bounds:
        features = [f for f in features if self._in_bounds(f, bounds)]

    center_lat, center_lon = self._calculate_center(features)

    buildings = {}
    outdoor_areas = {}
    raw_roads = []      # ← 新增

    for feature in features:
        geom_type = feature.get("geometry", {}).get("type", "")
        props = feature.get("properties", {})

        if geom_type == "Polygon":
            if self._is_building(props):
                building = self._extract_building(feature, center_lat, center_lon, scale)
                if building:
                    buildings[building.id] = building
            elif self._is_area(props):
                area = self._extract_outdoor_area(feature, center_lat, center_lon, scale)
                if area:
                    outdoor_areas[area.id] = area

        elif geom_type == "LineString":       # ← 新增
            if self._is_road(props):
                road = self._extract_road(feature, center_lat, center_lon, scale)
                if road:
                    raw_roads.append(road)

    # ← 新增: 道路分段
    street_segments = self._segment_roads(raw_roads, segment_length)
    for seg in street_segments:
        outdoor_areas[seg.id] = seg

    # ← 新增: Connection 推断
    connections = self._infer_connections(
        buildings, outdoor_areas, street_segments, building_snap_distance
    )

    # ... 计算 bounds, 返回 Region (同之前) ...
    return Region(..., connections=tuple(connections))
```

### 3b: 道路提取方法

```python
def _is_road(self, props: dict) -> bool:
    """检查是否为道路。"""
    highway = props.get("highway", "")
    # 排除不需要的类型
    excluded = {"motorway", "trunk", "construction", "proposed", "raceway"}
    return highway != "" and highway not in excluded

def _extract_road(self, feature, center_lat, center_lon, scale) -> dict | None:
    """提取道路为坐标点序列 + 元数据。"""
    coords = feature.get("geometry", {}).get("coordinates", [])
    if len(coords) < 2:
        return None

    points = [self._project(c[1], c[0], center_lat, center_lon, scale)
              for c in coords if len(c) >= 2]

    props = feature.get("properties", {})
    return {
        "points": points,
        "highway_type": props.get("highway", "road"),
        "name": props.get("name", ""),
        "osm_tags": {k: v for k, v in props.items()
                     if k in ("highway", "name", "surface", "sidewalk", "lit")},
    }
```

### 3c: 道路分段（核心新增）

```python
# 道路宽度映射
ROAD_WIDTH = {
    "footway": 2.0, "path": 2.0, "steps": 1.5,
    "residential": 6.0, "service": 4.0, "living_street": 5.0,
    "secondary": 8.0, "primary": 10.0, "tertiary": 7.0,
}

def _segment_roads(self, raw_roads: list[dict],
                   segment_length: float) -> list[OutdoorArea]:
    """将所有道路分段为 OutdoorArea。"""
    all_segments = []

    for road in raw_roads:
        points = road["points"]
        road_name = road["name"] or f"road_{self._road_counter}"
        road_id_base = road_name.lower().replace(" ", "_").replace("'", "")
        highway_type = road["highway_type"]
        width = ROAD_WIDTH.get(highway_type, 6.0)

        segments = self._split_line_into_segments(points, segment_length)

        for i, seg_points in enumerate(segments):
            seg_id = f"{road_id_base}_seg_{i}"
            polygon = self._line_to_polygon(seg_points, width)

            all_segments.append(OutdoorArea(
                id=seg_id,
                name=f"{road_name} ({i+1})" if road_name else seg_id,
                polygon=polygon,
                surface="asphalt",
                vegetation_density=0.0,
                area_type="street",
                osm_tags=road["osm_tags"],
                road_name=road_name,
                segment_index=i,
            ))

        self._road_counter += 1

    return all_segments

def _split_line_into_segments(self, points: list[Coord],
                               segment_length: float) -> list[list[Coord]]:
    """将点序列按长度分段。"""
    segments = []
    current = [points[0]]
    current_len = 0.0

    for i in range(1, len(points)):
        d = points[i-1].distance_to(points[i])
        current_len += d
        current.append(points[i])

        if current_len >= segment_length:
            segments.append(current)
            current = [points[i]]
            current_len = 0.0

    if len(current) >= 2:
        segments.append(current)
    elif segments and current:
        # 最后一段太短，合并到上一段
        segments[-1].extend(current[1:])

    return segments

def _line_to_polygon(self, line_points: list[Coord], width: float) -> Polygon:
    """将线段扩展为有宽度的多边形。"""
    half_w = width / 2
    left_side = []
    right_side = []

    for i in range(len(line_points)):
        if i < len(line_points) - 1:
            dx = line_points[i+1].x - line_points[i].x
            dy = line_points[i+1].y - line_points[i].y
        else:
            dx = line_points[i].x - line_points[i-1].x
            dy = line_points[i].y - line_points[i-1].y

        length = (dx**2 + dy**2) ** 0.5
        if length == 0:
            continue
        # 法线方向
        nx, ny = -dy / length, dx / length

        left_side.append(Coord(x=line_points[i].x + nx * half_w,
                               y=line_points[i].y + ny * half_w))
        right_side.append(Coord(x=line_points[i].x - nx * half_w,
                                y=line_points[i].y - ny * half_w))

    vertices = left_side + list(reversed(right_side))
    return Polygon(vertices=tuple(vertices))
```

### 3d: Connection 推断（核心新增）

```python
def _infer_connections(self, buildings, outdoor_areas, street_segments,
                       snap_distance) -> list[Connection]:
    """推断所有 Connection。"""
    connections = []

    # Step 1: 同一道路的相邻段串联
    roads = {}  # road_name -> sorted list of segments
    for seg in street_segments:
        roads.setdefault(seg.road_name, []).append(seg)

    for road_name, segs in roads.items():
        segs.sort(key=lambda s: s.segment_index)
        for i in range(len(segs) - 1):
            connections.append(Connection(
                from_id=segs[i].id,
                to_id=segs[i+1].id,
                path_type=segs[i].osm_tags.get("highway", "road"),
                distance=segs[i].center.distance_to(segs[i+1].center),
            ))

    # Step 2: 十字路口 — 不同道路的段端点靠近时连接
    all_seg_endpoints = []  # [(coord, segment)]
    for seg in street_segments:
        verts = seg.polygon.vertices
        start = verts[0]
        end = verts[len(verts)//2 - 1] if len(verts) > 2 else verts[-1]
        all_seg_endpoints.append((seg.center, seg))

    for i, (coord_i, seg_i) in enumerate(all_seg_endpoints):
        for j, (coord_j, seg_j) in enumerate(all_seg_endpoints[i+1:], i+1):
            if seg_i.road_name == seg_j.road_name:
                continue  # 同一条路的段已经在 Step 1 中连了
            dist = coord_i.distance_to(coord_j)
            if dist < snap_distance:
                connections.append(Connection(
                    from_id=seg_i.id,
                    to_id=seg_j.id,
                    path_type="intersection",
                    distance=dist,
                ))

    # Step 3: 建筑/公共空间 ↔ 最近的街道段
    all_locations = list(buildings.values()) + [
        a for a in outdoor_areas.values() if a.area_type != "street"
    ]
    for loc in all_locations:
        loc_center = loc.center if hasattr(loc, "center") else loc.polygon.center
        # 找最近的街道段
        nearest_seg = None
        nearest_dist = float("inf")
        for seg in street_segments:
            d = loc_center.distance_to(seg.center)
            if d < nearest_dist:
                nearest_dist = d
                nearest_seg = seg

        if nearest_seg and nearest_dist < snap_distance * 2:
            connections.append(Connection(
                from_id=loc.id,
                to_id=nearest_seg.id,
                path_type="entrance",
                distance=nearest_dist,
            ))

        # 如果有多个方向靠近的街道段，也连上（如公园有多个入口）
        for seg in street_segments:
            d = loc_center.distance_to(seg.center)
            if d < snap_distance and seg != nearest_seg:
                connections.append(Connection(
                    from_id=loc.id,
                    to_id=seg.id,
                    path_type="entrance",
                    distance=d,
                ))

    # Step 4: 兜底 — 没有连接到任何街道的 location，连到最近的 location
    connected_ids = {c.from_id for c in connections} | {c.to_id for c in connections}
    all_ids = set(buildings.keys()) | set(outdoor_areas.keys())
    isolated = all_ids - connected_ids

    for iso_id in isolated:
        iso_loc = buildings.get(iso_id) or outdoor_areas.get(iso_id)
        if not iso_loc:
            continue
        iso_center = iso_loc.center if hasattr(iso_loc, "center") else iso_loc.polygon.center
        # 连到最近的任意 location
        best_id, best_dist = None, float("inf")
        for other_id in (all_ids - {iso_id}):
            other_loc = buildings.get(other_id) or outdoor_areas.get(other_id)
            if not other_loc:
                continue
            other_center = other_loc.center if hasattr(other_loc, "center") else other_loc.polygon.center
            d = iso_center.distance_to(other_center)
            if d < best_dist:
                best_dist, best_id = d, other_id
        if best_id:
            connections.append(Connection(
                from_id=iso_id, to_id=best_id,
                path_type="direct", distance=best_dist,
            ))

    return connections
```

## 不需要改动的部分

### engine/simulation.py — 不改

SimulationService 的 `move_entity()` 接收 `to_location: str`（location ID），街道段就是一个 OutdoorArea 的 ID，完全兼容。

### engine/navigation.py — 不改

NavigationService 基于 Connection 图做 A* 寻路。街道段作为图中的节点，agent 从建筑到建筑的路径会自动经过中间的街道段。

### perception/ — 不改

感知管线基于 location ID 查询。agent 在街道段上时，能看到同一段的其他 agent。

### ledger/ — 不改

Ledger 存储 entity 的 location_id，不关心这个 ID 指向什么类型。
